Counter — Source
let mut count = signal(0);

view! {
    <div>
        <p class="big-num">{count}</p>
        <div class="row">
            <button on:click={move |_| count -= 1}>"−"</button>
            <button on:click={move |_| count.set(0)}>"Reset"</button>
            <button on:click={move |_| count += 1}>"+"</button>
        </div>
    </div>
}

Todo List — Source
let todos = signal(vec![...]);

view! {
    <input bind:value={input} />
    <ul>
        {for (text, done) in todos.get() {
